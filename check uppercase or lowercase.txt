#include <stdio.h>

int main() {
    char ch;

    printf("Enter a character: ");
    scanf(" %c", &ch);

    if (ch >= 'A' && ch <= 'Z')
        printf("The character is Uppercase");
    else if (ch >= 'a' && ch <= 'z')
        printf("The character is Lowercase");
    else
        printf("It is not an alphabet");

    return 0;
}
